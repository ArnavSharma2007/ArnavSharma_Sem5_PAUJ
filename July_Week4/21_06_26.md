## File Handling / Exceptions
### Custom Exceptions
For custom exceptions, we create a new class that extemds the parent exception class with 
only the super method being called inside of it to use the constructor.  

After that we just have to write "throws exceptioName" after the declaration of main.  

```Java
import java.io.*;

public class Main {
    static class Student{
        private String name;
        private int marks;

        Student(String name, int marks){
            this.name = name;
            this.marks = marks;
        }

        public String getName(){
            return name;
        }
        
        public int getMarks(){
            return marks;
        }
    }

    static class Box<T>{
        private T value;
        public void set(T value){
            this.value = value;
        }

        public T get(){
            return value;
        }
    }

    public static void saveStudent(Student s)
        throws IOException, ageNotFoundException{
            FileWriter writer = new FileWriter("Student.txt", true);
            writer.write("Name: " + s.getName() + " \nMarks: " + s.getMarks());
            writer.write("\n");
            writer.close();
        }

    public static void main(String args[]) {
        Box<Student> box = new Box<>();
        Student student = new Student("Arnav", 99);
        box.set(student);
        try{
            Student s = box.get();
            if(s.getMarks() < 0 || s.getMarks() > 100){
                throw new RuntimeException("Invalid Marks");
            }
            saveStudent(s);
            System.out.println("Student saved successfully");
            System.out.println("Name: " + s.getName());
            System.out.println("Marks: " + s.getMarks());
        }
        catch (IOException e){
            System.out.println(e);
        }
        catch (Exception e){
            System.out.println(e);
        }
        finally{
            System.out.println("Program has finished execution.");
        }
    }

    static class ageNotFoundException extends Exception {
        public ageNotFoundException(String message) {
            super(message);
        }
    }
}

```
